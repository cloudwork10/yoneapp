```javascript
  quarterly: {
    name: 'Quarterly Plan',
    price: 199, // 199 EGP or $4
    duration: 90, // days
    discount: 33 // (297-199)/297 * 100
  },
  'semi-annual': {
    name: 'Semi-Annual Plan',
    price: 299, // 299 EGP or $6
    duration: 180, // days
    discount: 50 // (594-299)/594 * 100
  },
  annual: {
    name: 'Annual Plan',

    duration: 365, // days
    discount: 58 // (1188-499)/1188 * 100
  }
};
```
